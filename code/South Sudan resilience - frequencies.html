    pivot_longer(cols=1:20,
               names_to="income_source",
               values_to="Percent") %>%
  mutate(inc_lab=inc_labs) %>%
  relocate(inc_lab, .after=income_source)

inc_ov_se <- inc_ov_tmp[,ov_cols==F] %>%

  bind_cols(inc_ov_se[,2]) %>%

  select(-1, -4) %>%

  gt() %>%
  fmt_percent(2:5,
