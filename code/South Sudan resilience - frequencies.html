  set_names(nm=c("income_source","mean","se")) %>%
  mutate(margin=se*1.96,
         lower=mean-1.96*margin,
         upper=mean+1.96*margin)

inc_gt <- inc_frq %>%
  arrange(desc(mean)) %>%
  gt() %>%
  fmt_percent(2:6,
